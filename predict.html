‚Üê Make Another Prediction Option 2: Streamlit App (Easiest) Create streamlit_app.py: pythonimport streamlit as st import joblib import pandas as pd import numpy as np # Load model @st.cache_resource def load_model(): return joblib.load('crop_yield_model.pkl') model = load_model() st.title('üåæ Crop Yield Predictor') st.write('Enter the parameters below to predict crop yield:') # Input fields area = st.text_input('Area', placeholder='e.g., Punjab') crop = st.text_input('Crop', placeholder='e.g., Wheat') soil_type = st.text_input('Soil Type', placeholder='e.g., Loamy') temperature = st.number_input('Temperature (¬∞C)', min_value=0.0, max_value=50.0, value=25.0) humidity = st.number_input('Humidity (%)', min_value=0.0, max_value=100.0, value=50.0) ph = st.number_input('pH Level', min_value=0.0, max_value=14.0, value=7.0) rainfall = st.number_input('Annual Rainfall (mm)', min_value=0.0, value=800.0) if st.button('Predict Yield'): if area and crop and soil_type: input_data = { 'Area': area, 'Crop': crop, 'Soil_Type': soil_type, 'Temperature': temperature, 'Humidity': humidity, 'PH': ph, 'Annual_Rainfall': rainfall } input_df = pd.DataFrame([input_data]) prediction = model.predict(input_df)[0] st.success(f'üéØ Predicted Yield: {prediction:.2f} tons/hectare') # Show input summary st.subheader('Input Summary:') st.write(input_data) else: st.error('Please fill in all required fields!')
